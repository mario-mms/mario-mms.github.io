Proyecto modelo relacional

Para crear una base de datos desde cero, debemos seguir el siguiente procedimiento:

  1. Analizar los datos que vamos a tener que introducir
  2. Dibujar el esquema de la base de datos con el modelo relacional. (Acercar ratón para ver el esquema)
  3. Crear las tablas con sus nombres, columnas, tipos de datos y restricciones apropiadas. create table canales ( id_canal number(2) constraint canales_pk1 primary key , nombre varchar2(50) constraint canales_uk1 unique constraint canales_nn1 not null ); create table categoria( id_categoria number(1) constraint categoria_pk1 primary key , recomendado varchar2(3) constraint categoria_uk1 unique constraint categoria_nn1 not null ); create table programas( id_programa number(5) constraint programas_pk1 primary key , id_canal number(2) constraint programas_fk1 references canales on delete cascade , nombre varchar2(50) constraint programas_nn1 not null, constraint programas_uk1 unique (id_canal,nombre) ); alter table programas modify ( id_canal constraint programas_nn3 not null ); create table capitulos( id_emision number(10) constraint capitulos_pk1 primary key , id_programa number(5) constraint capitulos_fk1 references programas on delete cascade , n_capitulo number(5), titulo varchar2(100), descripcion varchar2(200), id_categoria number(1) constraint programas_fk2 references categoria ); alter table capitulos modify ( id_programa constraint capitulos_nn1 not null , id_categoria constraint capitulos_nn2 not null ); create table emision( id_emision number(10) constraint emision_pk1 primary key constraint emision_fk1 references capitulos(id_emision) on delete cascade, fecha_inicio timestamp constraint emision_nn1 not null , fecha_fin timestamp constraint emision_nn2 not null );
